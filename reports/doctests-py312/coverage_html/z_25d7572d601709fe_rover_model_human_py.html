155class Percieve(Action, GenericHumanAction): 

156 """ 

157 Operator percieving the state of comms signals, switches, etc. 

158 

159 Should take in view (from look) and pass as percieved info. 

160 

161 Examples 

162 -------- 

163 >>> p = Percieve() 

164 >>> p.comms.s.on = True 

165 >>> p.comms.s.pos.x = 1.0 

166 >>> p.comms.s.video.lin_ux = 2.0 

167 >>> p.behavior(1) 

168 >>> p.pos_signal.s.x 

169 1.0 

170 >>> p.video.s.lin_ux 

171 2.0 

172 """ 

173 

174 __slots__ = ('comms', 'pos_signal', 'video', 'psfs') 

175 container_m = PerceptionMode 

176 flow_comms = Comms 

177 flow_pos_signal = Pos 

178 flow_video = Video 

179 flow_psfs = PSFs 

180 

181 def behavior(self, t): 

182 if not self.comms.s.on: 

183 self.m.remove_any_faults(opermode='no_action') 

184 elif self.comms.s.video.quality == 0.0: 

185 self.m.to_fault('not_visible') 

186 elif self.psfs.p.fatigue > 8: 

187 self.m.to_fault("failed_no_action") 

188 elif self.psfs.s.attention < 3: 

189 self.m.to_fault("wrong_position") 

190 else: 

191 self.m.remove_any_faults(opermode='nominal') 

192 

193 if not self.m.in_mode("not_visible", 'no_action'): 

194 self.video.s.assign(self.comms.s.video) 

195 if not self.m.in_mode("wrong_position", 'no_action'): 

196 self.pos_signal.s.assign(self.comms.s.pos)

199class Comprehend(Action, GenericHumanAction): 

200 """ 

201 Operator comprehending the input state. 

202 

203 Should take in percieved info and distill as situation (moving, turning, etc) 

204 

205 May fail due to fatigue >8 or stress >80. 

206 

207 Examples 

208 -------- 

209 >>> c = Comprehend() 

210 >>> c.pos_signal.s.put(ux=1.0, uy=1.0) 

211 >>> c.behavior(1) 

212 >>> c.signal.s.u_self 

213 array([1., 1.]) 

214 """ 

215 

216 __slots__ = ('signal', 'pos_signal', 'video', 'psfs') 

217 container_m = HumanActionMode 

218 flow_signal = OperatorSignal 

219 flow_pos_signal = Pos 

220 flow_video = Video 

221 flow_psfs = PSFs 

222 

223 def behavior(self, t): 

224 if self.psfs.p.fatigue > 8 or self.psfs.p.stress > 80: 

225 self.m.to_fault('failed_no_action') 

226 if self.m.in_mode('nominal'): 

227 self.signal.s.set_positions(self.pos_signal, self.video)

252class Project(Action, GenericHumanAction): 

253 """ 

254 Operator projecting out how control actions might affect input state. 

255 

256 Determines rdiff in signal. 

257 

258 Examples 

259 -------- 

260 >>> p = Project() 

261 >>> p.signal.s.put(u_self=(0.0, 1.0), u_lin=(1.0, 1.0)) 

262 >>> p.behavior(1) 

263 >>> p.signal.s.rdiff 

264 -0.7854052343902613 

265 """ 

266 

267 __slots__ = ('signal', 'psfs') 

268 container_m = ProjectMode 

269 flow_signal = OperatorSignal 

270 flow_psfs = PSFs 

271 

272 def behavior(self, t): 

273 if self.m.in_mode('nominal'): 

274 self.signal.s.set_turn() 

275 elif self.m.in_mode('failed_turn_left'): 

276 self.signal.s.rdiff = -0.5 

277 elif self.m.in_mode('failed_turn_right'): 

278 self.signal.s.rdiff = 0.5 

279 elif self.m.in_mode('failed_slow'): 

280 self.signal.s.set_turn() 

281 self.signal.s.vel_adj = 0.2 

282 elif self.m.in_mode('failed_fast'): 

283 self.signal.s.set_turn() 

284 self.signal.s.vel_adj = 4.0 

285 # increment attention - degrades over the course of an operation if the 

286 # driving isn't "interesting" (aka there are no turns) 

287 # turns reset the attention to 10 if attention is above threshold 

288 if abs(self.signal.s.rdiff) < 0.01: 

289 if self.psfs.p.fatigue < 5: 

290 self.psfs.s.inc(attention=(-0.2, 0.0)) 

291 else: 

292 self.psfs.s.inc(attention=(-0.4, 0.0)) 

293 elif self.psfs.s.attention > 3: 

294 self.psfs.s.attention = 10.0

363class Press(Action, GenericHumanAction): 

364 """ 

365 Operator presses the button/toggle for the controls. 

366 

367 Examples 

368 -------- 

369 >>> p = Press() 

370 >>> p.control.s.put(lpower=1.0, rpower=0.0) 

371 >>> p.behavior(1) 

372 >>> p.comms.s.ctl 

373 ControlState(rpower=0.0, lpower=1.0) 

374 """ 

375 

376 __slots__ = ('control', 'comms') 

377 container_m = PressMode 

378 flow_control = Control 

379 flow_comms = Comms 

380 

381 def behavior(self, t): 

382 if self.m.in_mode('nominal'): 

383 self.comms.s.ctl.assign(self.control.s) 

384 elif self.m.in_mode('failed_left'): 

385 self.comms.s.ctl.put(lpower=1.0, rpower=0.0) 

386 elif self.m.in_mode('failed_right'): 

387 self.comms.s.ctl.put(lpower=0.0, rpower=1.0) 

388 elif self.m.in_mode('no_press'): 

389 self.comms.s.ctl.put(lpower=0.0, rpower=0.0)

392class HumanActions(ActionArchitecture): 

393 """ 

394 Overall ASG for human operator driving the rover. 

395 

396 Examples 

397 -------- 

398 >>> ha = HumanActions() 

399 >>> ha.flows['comms'].s.on = True 

400 >>> ha.active_actions 

401 {'look'} 

402 >>> ha('dynamic', 1, False, 1) 

403 >>> ha.active_actions 

404 {'press'} 

405 """ 

406 __slots__ = () 

407 initial_action = "look" 

408 per_timestep = True 

409 container_p = PSFParam 

410 

411 def init_architecture(self, **kwargs): 

412 self.add_flow('psfs', PSFs, p=self.p) 

413 self.add_flow('signal', OperatorSignal) 

414 self.add_flow('pos_signal', Pos) 

415 self.add_flow('switch', Switch) 

416 self.add_flow('control', Control) 

417 self.add_flow('comms', Comms) 

418 self.add_flow('video', Video) 

419 # flow for workload "local PSF"? 

420 

421 self.add_act('look', Look) 

422 self.add_act('percieve', Percieve, 'comms', 'pos_signal', 'video', 'psfs') 

423 self.add_act('comprehend', Comprehend, 'pos_signal', 'video', 'signal', 'psfs') 

424 self.add_act('project', Project, 'signal', 'psfs') 

425 self.add_act('decide', Decide, 'signal', 'control') 

426 self.add_act('reach', Reach) 

427 self.add_act('press', Press, 'comms', 'control') 

428 

429 self.add_cond('look', 'percieve', condition=self.acts['look'].complete) 

430 self.add_cond('percieve', 'comprehend', 

431 condition=self.acts['percieve'].complete) 

432 self.add_cond('comprehend', 'project', 

433 condition=self.acts['comprehend'].complete) 

434 self.add_cond('project', 'decide', condition=self.acts['project'].complete) 

435 self.add_cond('decide', 'reach', condition=self.acts['decide'].complete) 

436 self.add_cond('reach', 'press', condition=self.acts['reach'].complete)

440 """Overall function for operator (adds ASG to flipping switch).""" 

441 

442 __slots__ = ('switch', 'control', 'comms', 'pos_signal', 'video', 'psfs', 'ground')

444 flow_control = Control 

445 flow_pos_signal = Pos 

446 flow_comms = Comms

448 flow_psfs = PSFs 

449 flow_ground = Ground 

450 container_p = PSFParam 

451 arch_aa = HumanActions 

452 container_m = Mode

454 def dynamic_behavior(self, t): 

455 self.set_power(t) 

456 if self.ground.at_end(self.pos_signal.s): 

457 self.switch.s.power = False 

458 self.comms.s.ctl.put(rpower=0, lpower=0)
