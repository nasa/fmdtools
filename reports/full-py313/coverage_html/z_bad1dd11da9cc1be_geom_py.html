34class GeomArchitecture(Architecture): 

35 """ 

36 Agglomeration of multiple geoms/shapes. 

37 

38 Architecture is defined using add_shape method in user-defined init_shapes method. 

39 

40 Examples 

41 -------- 

42 for an architecture with the geoms already defined: 

43 

44 >>> class ExGeomArch(GeomArchitecture): 

45 ... def init_architecture(self): 

46 ... self.add_point("ex_point", ExPoint) 

47 ... self.add_line("ex_line", ExLine) 

48 ... self.add_poly("ex_poly", ExPoly) 

49 ... def dynamic_behavior(self): 

50 ... if self.t.time >= 1.0: 

51 ... self.points["ex_point"].s.buffer_around = self.t.time 

52 ... def static_behavior(self): 

53 ... self.lines["ex_line"].s.buffer_around = self.points["ex_point"].s.buffer_around 

54 

55 This can then be used in containing classes (e.g., environments) that need multiple 

56 geoms. We can then access the individual geoms in the geoms dict, e.g.,: 

57 

58 >>> ega = ExGeomArch() 

59 >>> ega 

60 exgeomarch ExGeomArch 

61 - t=Time(time=-0.1, timers={}) 

62 POINTS: 

63 - ex_point=ExPoint(s=(occupied=False, buffer_around=1.0)) 

64 LINES: 

65 - ex_line=ExLine(s=(occupied=False, buffer_around=1.0)) 

66 POLYS: 

67 - ex_poly=ExPoly(s=(occupied=False, buffer_around=1.0)) 

68 >>> ega.geoms()['ex_point'].s 

69 ExGeomState(occupied=False, buffer_around=1.0) 

70 >>> ega.h 

71 time: array(101) 

72 points.ex_point.s.occupied: array(101) 

73 points.ex_point.s.buffer_around: array(101) 

74 lines.ex_line.s.occupied: array(101) 

75 lines.ex_line.s.buffer_around: array(101) 

76 polys.ex_poly.s.occupied: array(101) 

77 polys.ex_poly.s.buffer_around: array(101) 

78 >>> ega.return_mutables() 

79 ((False, 1.0), (False, 1.0), (False, 1.0), (-0.1, 0, False, False, False)) 

80 

81 GeomArchitectures are also simulable provided dynamic_behavior and static_behavior 

82 methods as shown below. Note that this behavior must be called externally, 

83 such as from a function, to have meaning in a broader modelling context. 

84 

85 >>> ega(time=2.0) 

86 >>> ega 

87 exgeomarch ExGeomArch 

88 - t=Time(time=2.0, timers={}) 

89 POINTS: 

90 - ex_point=ExPoint(s=(occupied=False, buffer_around=2.0)) 

91 LINES: 

92 - ex_line=ExLine(s=(occupied=False, buffer_around=2.0)) 

93 POLYS: 

94 - ex_poly=ExPoly(s=(occupied=False, buffer_around=1.0)) 

95 """

116 def check_geom_class(self, gclass, baseclass): 

117 """Check that geom class/object inherits from given base class.""" 

118 try: 

119 if not issubclass(gclass, baseclass): 

120 raise Exception("gclass "+gclass+" not a "+baseclass.__name__) 

121 except TypeError: 

122 if not isinstance(gclass, baseclass): 

123 raise Exception("gclass "+gclass+" not a "+baseclass.__name__)

173 def all_at(self, *pt): 

174 """ 

175 Find all geoms (and buffers) a given is at. 

176 

177 Parameters 

178 ---------- 

179 *pt : x,y 

180 x, y, z location to check. 

181 

182 Returns 

183 ------- 

184 all_at : dict 

185 Names of geoms where the point is at (and their properties) 

186 

187 Examples 

188 -------- 

189 >>> exga = ExGeomArch() 

190 >>> exga.all_at(1.0, 1.0) 

191 {'ex_point': ['shape', 'on', 'around'], 'ex_line': ['shape', 'on', 'around'], 'ex_poly': ['shape', 'around']} 

192 >>> exga.all_at(0.0, 0.0) 

193 {'ex_line': ['shape', 'on', 'around'], 'ex_poly': ['shape', 'around']} 

194 >>> exga.all_at(0.4, 0.3) 

195 {'ex_point': ['on', 'around'], 'ex_line': ['on', 'around'], 'ex_poly': ['around']} 

196 """ 

197 all_at = {} 

198 for geomname, geom in self.geoms().items(): 

199 at_geom = geom.all_at(*pt) 

200 if at_geom: 

201 all_at[geomname] = at_geom 

202 return all_at

204 def show(self, geoms={'all': {}}, fig=None, ax=None, figsize=(4, 4), z=False, 

205 **kwargs): 

206 """ 

207 Show the shapes of a GeomArchitecture all on one plot. 

208 

209 Parameters 

210 ---------- 

211 geoms : dict, optional 

212 Individual shapes to plot and their corresponding kwargs. 

213 The default is {'all': {}}. 

214 fig : matplotlib.figure, optional 

215 Existing Figure. The default is None. 

216 ax : matplotlib.axis, optional 

217 Existing axis. The default is None. 

218 figsize : tuple, optional 

219 Size for figure (if instantiating). The default is (4, 4). 

220 z : bool/number, optional 

221 If plotting on a 3d axis, set z to a number which will be the z-level. 

222 The default is False. 

223 **kwargs : kwargs 

224 Overall kwargs to show.geom for all geoms. 

225 

226 Returns 

227 ------- 

228 fig : figure 

229 Matplotlib figure object 

230 ax : axis 

231 Corresponding matplotlib axis 

232 """ 

233 if not ax: 

234 fig, ax = setup_plot(z=z, figsize=figsize) 

235 if 'all' in geoms: 

236 geoms = {g: {'shapes': 'all'} for g in self.geoms()} 

237 

238 for geomname, geom_kwargs in geoms.items(): 

239 local_kwargs = {**kwargs, 'geomlabel': geomname, **geom_kwargs} 

240 fig, ax = self.geoms()[geomname].show(ax=ax, fig=fig, z=z, **local_kwargs) 

241 return fig, ax

243 def show_from(self, hist, t, **kwargs): 

244 """ 

245 Show the GeomArch at the given time in the provided history. 

246 

247 Parameters 

248 ---------- 

249 hist : History 

250 History of states for the GeomArchitecture. 

251 t : int 

252 Time in the history to show the state of the GeomArchitecture at. 

253 **kwargs : kwargs 

254 Keyword arguments to GeomArchitecture.show(). 

255 

256 Returns 

257 ------- 

258 fig : figure 

259 Matplotlib figure object 

260 ax : axis 

261 Corresponding matplotlib axis 

262 """ 

263 for flex_role in self.flexible_roles: 

264 for geomname, geom in self.get_flex_role_objs(flex_role).items(): 

265 geom.assign_from(hist.get(flex_role+"s."+geomname), t) 

266 return self.show(**kwargs)
