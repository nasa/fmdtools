18class GeomArchitecture(Architecture): 

19 """ 

20 Agglomeration of multiple geoms/shapes. 

21 

22 Architecture is defined using add_shape method in user-defined init_shapes method. 

23 

24 Examples 

25 -------- 

26 for an architecture with the geoms already defined: 

27 

28 >>> class ExGeomArch(GeomArchitecture): 

29 ... def init_architecture(self): 

30 ... self.add_point("ex_point", ExPoint) 

31 ... self.add_line("ex_line", ExLine) 

32 ... self.add_poly("ex_poly", ExPoly) 

33 

34 This can then be used in containing classes (e.g., environments) that need multiple 

35 geoms. We can then access the individual geoms in the geoms dict, e.g.,: 

36 

37 >>> ega = ExGeomArch() 

38 >>> ega.geoms()['ex_point'].s 

39 ExGeomState(occupied=False) 

40 >>> ega.h 

41 points.ex_point.s.occupied: array(101) 

42 lines.ex_line.s.occupied: array(101) 

43 polys.ex_poly.s.occupied: array(101) 

44 >>> ega.return_mutables() 

45 ((False,), (False,), (False,), (-0.1, 0, 0.0, 1)) 

46 """

66 def check_geom_class(self, gclass, baseclass): 

67 """Check that geom class/object inherits from given base class.""" 

68 try: 

69 if not issubclass(gclass, baseclass): 

70 raise Exception("gclass "+gclass+" not a "+baseclass.__name__)

123 def all_at(self, *pt): 

124 """ 

125 Find all geoms (and buffers) a given is at. 

126 

127 Parameters 

128 ---------- 

129 *pt : x,y 

130 x, y, z location to check. 

131 

132 Returns 

133 ------- 

134 all_at : dict 

135 Names of geoms where the point is at (and their properties) 

136 

137 Examples 

138 -------- 

139 >>> exga = ExGeomArch() 

140 >>> exga.all_at(1.0, 1.0) 

141 {'ex_point': ['shape', 'on'], 'ex_line': ['shape', 'on'], 'ex_poly': ['shape']} 

142 >>> exga.all_at(0.0, 0.0) 

143 {'ex_line': ['shape', 'on'], 'ex_poly': ['shape']} 

144 >>> exga.all_at(0.4, 0.3) 

145 {'ex_point': ['on'], 'ex_line': ['on']} 

146 """ 

147 all_at = {} 

148 for geomname, geom in self.geoms().items(): 

149 at_geom = geom.all_at(*pt) 

150 if at_geom: 

151 all_at[geomname] = at_geom 

152 return all_at

154 def show(self, geoms={'all': {}}, fig=None, ax=None, figsize=(4, 4), z=False, 

155 **kwargs): 

156 """ 

157 Show the shapes of a GeomArchitecture all on one plot. 

158 

159 Parameters 

160 ---------- 

161 geoms : dict, optional 

162 Individual shapes to plot and their corresponding kwargs. 

163 The default is {'all': {}}. 

164 fig : matplotlib.figure, optional 

165 Existing Figure. The default is None. 

166 ax : matplotlib.axis, optional 

167 Existing axis. The default is None. 

168 figsize : tuple, optional 

169 Size for figure (if instantiating). The default is (4, 4). 

170 z : bool/number, optional 

171 If plotting on a 3d axis, set z to a number which will be the z-level. 

172 The default is False. 

173 **kwargs : kwargs 

174 Overall kwargs to show.geom for all geoms. 

175 

176 Returns 

177 ------- 

178 fig : figure 

179 Matplotlib figure object 

180 ax : axis 

181 Corresponding matplotlib axis 

182 """ 

183 if not ax: 

184 fig, ax = setup_plot(z=z, figsize=figsize) 

185 if 'all' in geoms: 

186 geoms = {g: {'shapes': 'all'} for g in self.geoms()} 

187 

188 for geomname, geom_kwargs in geoms.items(): 

189 local_kwargs = {**kwargs, 'geomlabel': geomname, **geom_kwargs} 

190 fig, ax = self.geoms()[geomname].show(ax=ax, fig=fig, z=z, **local_kwargs) 

191 return fig, ax
