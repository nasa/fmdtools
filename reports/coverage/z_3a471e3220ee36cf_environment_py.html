9class Environment(CommsFlow): 

10 """ 

11 Class for representing environments (in development). 

12 

13 Environments are CommsFlows in order to readily enable perception as well as 

14 sending and recieving of information. In addition to having normal flow properties, 

15 they also contain the roles: 

16 

17 TODO: Properly expand create_local, update, send, recieve, etc to use ga and coords. 

18 

19 Roles 

20 --------------- 

21 c: Coords 

22 Representation of gridworld properties 

23 r: Rand 

24 Representation of random variables/rng 

25 ga: GeomArch 

26 Representaion of shapes/forms 

27 

28 Examples 

29 -------- 

30 >>> class ExampleEnvironment(Environment): 

31 ... coords_c = ExampleCoords 

32 ... arch_ga = ExGeomArch 

33 >>> env = ExampleEnvironment('env') 

34 >>> env.create_hist([1.0]) 

35 c.r.probdens: array(1) 

36 c.st: array(1) 

37 ga.points.ex_point: array(1) 

38 ga.lines.ex_line: array(1) 

39 ga.polys.ex_poly: array(1) 

40 """

73 def copy(self, glob=[], p={}, s={}): 

74 """ 

75 Copy the Environment. 

76 

77 Examples 

78 -------- 

79 Copies should be identical but independent after copying, e.g.: 

80 

81 >>> e = ExampleEnvironment("env") 

82 >>> e.ga.points['ex_point'].s.occupied = True 

83 >>> e.c.st[0, 0] = 1 

84 

85 Given these changes, the copy should have the same states (and not default): 

86 

87 >>> d = e.copy() 

88 >>> d.ga.points['ex_point'].s.occupied 

89 True 

90 >>> d.c.st[0, 0] 

91 1.0 

92 

93 It should also be independent, meaning changes don't effect the original: 

94 

95 >>> d.c.st[0, 1] = 1.0 

96 >>> e.c.st[0, 1] 

97 0.0 

98 >>> d.ga.lines['ex_line'].s.occupied = True 

99 >>> e.ga.lines['ex_line'].s.occupied 

100 False 

101 """ 

102 cop = super().copy(glob=glob, p=p, s=s) 

103 cop.r.assign(self.r) 

104 cop.c = self.c.copy() 

105 cop.ga = self.ga.copy() 

106 if hasattr(self, 'h'): 

107 cop.h = self.h.copy() 

108 return cop
