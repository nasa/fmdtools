13class ActionArchitecture(Architecture): 

14 """ 

15 Construct the Action Sequence Graph with the given parameters. 

16 

17 Parameters 

18 ---------- 

19 initial_action : str/list 

20 Initial action to set as active. Default is 'auto' 

21 - 'auto' finds the starting node of the graph and uses it 

22 - 'ActionName' sets the given action as the first active action 

23 - providing a list of actions will set them all to active 

24 (if multi-state rep is used) 

25 state_rep : 'finite-state'/'multi-state' 

26 How the states of the system are represented. Default is 'finite-state' 

27 - 'finite-state' means only one action in the system can be active at once 

28 (i.e., a finite state machine) 

29 - 'multi-state' means multiple actions can be performed at once 

30 max_action_prop : 'until_false'/'manual'/int 

31 How actions progress. Default is 'until_false' 

32 - 'until_false' means actions are simulated until all outgoing conditions 

33 are false 

34 - providing an integer places a limit on the number of actions that can be 

35 performed per timestep 

36 proptype : 'static'/'dynamic'/'manual' 

37 Which propagation step to execute the Action Sequence Graph in. 

38 Default is 'dynamic' 

39 - 'manual' means that the propagation is performed manually 

40 (defined in a behavior method) 

41 per_timestep : bool 

42 Defines whether the action sequence graph is reset to the initial state each 

43 time-step (True) or stays in the current action (False). Default is False. 

44 """ 

45 __slots__ = ['acts', 'action_graph', 'flow_graph', 'faultmodes', 'active_actions'] 

46 initial_action = "auto" 

47 state_rep = "finite-state" 

48 max_action_prop = "until_false" 

49 proptype = 'dynamic' 

50 per_timestep = False 

51 default_track = ('acts', 'flows', 'active_actions', 'i') 

52 flexible_roles = ['flows', 'acts', 'conds']

168 def __call__(self, proptype, time, run_stochastic, dt): 

169 """ 

170 Propagates behaviors through the ActionArchitecture. 

171 

172 Parameters 

173 ---------- 

174 time : float, optional 

175 Model time. The default is 0. 

176 run_stochastic : bool/str 

177 Whether to run the simulation using stochastic or deterministic behavior 

178 proptype : str 

179 Type of propagation step to update 

180 ('behavior', 'static_behavior', or 'dynamic_behavior') 

181 dt : float 

182 Timestep to propagate over. 

183 """ 

184 if self.per_timestep: 

185 self.set_active_actions(self.initial_action) 

186 for action in self.active_actions: 

187 self.acts[action].t.t_loc = 0.0 

188 

189 if proptype == self.proptype: 

190 active_actions = self.active_actions 

191 num_prop = 0 

192 while active_actions: 

193 new_active_actions = set(active_actions) 

194 for action in active_actions: 

195 self.acts[action](time, run_stochastic, proptype=proptype, dt=dt) 

196 action_cond_edges = self.action_graph.out_edges(action, data=True) 

197 for act_in, act_out, atts in action_cond_edges: 

198 try: 

199 cond = self.conds[atts['name']]() 

200 except TypeError as e: 

201 raise TypeError("Poorly specified condition " + 

202 str(atts['name'])+": ") from e 

203 if cond and getattr(self.acts[action], 'duration', 0.0)+dt <= self.acts[action].t.t_loc: 

204 self.acts[action].t.t_loc = 0.0 

205 new_active_actions.add(act_out) 

206 new_active_actions.discard(act_in) 

207 if len(new_active_actions) > 1 and self.state_rep == 'finite-state': 

208 raise Exception("Multiple active actions in a finite-state " + 

209 "representation: "+str(new_active_actions)) 

210 num_prop += 1 

211 if type(self.proptype) == int and num_prop >= self.proptype: 

212 break 

213 if new_active_actions == set(active_actions): 

214 break 

215 else: 

216 active_actions = new_active_actions 

217 if num_prop > 10000: 

218 raise Exception("Undesired looping in Function ASG for: "+self.name) 

219 self.active_actions = active_actions
