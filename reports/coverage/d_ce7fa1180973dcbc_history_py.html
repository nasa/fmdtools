39def init_hist_iter(att, val, timerange=None, track=None, dtype=None, str_size='<U20'): 

40 """ 

41 Initialize the history for a given attribute att with value val. 

42 

43 Enables the recursive definition of a history as a nested structure. 

44 

45 If a timerange is given, the base fields are initializes as fixed-length numpy 

46 arrays corresponding to the data type of the field. Otherwise, an empty list 

47 is initialized. 

48 

49 Parameters 

50 ---------- 

51 att : str 

52 Name of the attribute. 

53 val : dict/field 

54 dict to be initialized as a History or field to be initialized as a list or 

55 numpy array 

56 timerange : iterable, optional 

57 Time-range to initialize the history over. The default is None. 

58 track : list/str/dict, optional 

59 argument specifying attributes for :func:`get_sub_include'. The default is None. 

60 dtype : str, optional 

61 Datatype to initialze the array as (if given). The default is None. 

62 str_size : str, optional 

63 Data type for strings. The default is '<U20'. 

64 

65 Returns 

66 ------- 

67 Hist : History, List, or np.array 

68 Initialized history structure corresponding to the attribute 

69 """ 

70 sub_track = get_sub_include(att, track) 

71 if sub_track and hasattr(val, 'create_hist'): 

72 return val.create_hist(timerange, sub_track) 

73 elif sub_track and isinstance(val, dict): 

74 return init_dicthist(val, timerange, sub_track) 

75 elif sub_track: 

76 hist = History() 

77 hist.init_att(att, val, timerange, track, dtype, str_size)

104def diff(val1, val2, difftype='bool'): 

105 """ 

106 Find inconsistent states between val1, val2. 

107 

108 The difftype option ('diff' (takes the difference), 'bool' (checks if the same), and float (checks if under the provided tolerance)) 

109 """ 

110 if difftype == 'diff': 

111 return val1-val2 

112 elif difftype == 'bool': 

113 return val1 != val2 

114 elif type(difftype) == float: 

115 return abs(val1-val2) > difftype

145class History(Result): 

146 """ 

147 Class for recording and analyzing simulation histories. 

148 

149 Histories log states of the model over time. 

150 

151 It can be updated over time t using h.log(obj, t), where obj is an object with 

152 (nested) attributes that match the keys of the (nested) dictionary. 

153 

154 Examples 

155 -------- 

156 # histories act the same as results, but with the values being arrays: 

157 

158 >>> hist = History({"a": [1, 2, 3], "b": [4, 5, 6], "time": [0, 1, 2]}) 

159 >>> hist 

160 a: array(3) 

161 b: array(3) 

162 time: array(3) 

163 

164 history access is the same as a result: 

165 

166 >>> hist.a 

167 [1, 2, 3] 

168 

169 metrics can be gotten from histories over time: 

170 

171 >>> hist = History({"a.a": [1, 2, 3], "b.a": [4, 5, 6], "time": [0, 1, 2]}) 

172 >>> hist.get_metric("a", axis=0) 

173 array([2.5, 3.5, 4.5]) 

174 

175 or over all times: 

176 

177 >>> hist.get_metric("a") 

178 3.5 

179 """

181 def init_att(self, att, val, 

182 timerange=None, track=None, dtype=None, str_size='<U20'): 

183 sub_track = get_sub_include(att, track) 

184 if sub_track: 

185 if type(val) == dict or (hasattr(val, 'keys') and hasattr(val, 'values')): 

186 self[att] = init_dicthist(val, timerange, sub_track) 

187 elif timerange is None: 

188 self[att] = [val] 

189 elif type(val) == str: 

190 self[att] = np.empty([len(timerange)], dtype=str_size) 

191 elif type(val) == np.ndarray or dtype == np.ndarray: 

192 self[att] = np.array([val for i in timerange]) 

193 elif dtype: 

194 self[att] = np.empty([len(timerange)], dtype=dtype) 

195 else: 

196 try: 

197 self[att] = np.full(len(timerange), val) 

198 except: 

199 self[att] = np.empty((len(timerange),), dtype=object)

251 def log(self, obj, t_ind, time=None): 

252 """ 

253 Update the history from obj at the time t_ind. 

254 

255 Parameters 

256 ---------- 

257 obj : Model/Function/State... 

258 Object to log 

259 t_ind : int 

260 Time-index of the log. 

261 time : float 

262 Real time for the history (if initialized). Used at the top level of the 

263 history. 

264 """ 

265 for att, hist in self.items(): 

266 try: 

267 val = None 

268 if att == 'time' and time is not None: 

269 val = time 

270 elif att.startswith('i.') or '.i.' in att: 

271 split_att = att.split('.') 

272 i_ind = split_att.index('i') 

273 new_split_att = split_att[:i_ind] + ['indicate_'+split_att[-1]] 

274 methname = '.'.join(new_split_att) 

275 try: 

276 val = get_var(obj, methname)(time) 

277 except TypeError: 

278 val = get_var(obj, methname)() 

279 elif 'faults' in att: 

280 split_att = att.split('.') 

281 faultind = split_att.index('faults') 

282 modename = split_att[faultind+1] 

283 fault_att = '.'.join(split_att[:faultind]) 

284 val = modename in get_var(obj, fault_att).faults 

285 else: 

286 val = get_var(obj, att) 

287 except: 

288 raise Exception("Unable to log att " + str(att) + " in " + 

289 str(obj.__class__.__name__) + ', val=' + str(val)) 

290 

291 if type(hist) == History: 

292 hist.log(val, t_ind) 

293 else: 

294 if is_known_mutable(val): 

295 val = copy.deepcopy(val) 

296 if type(hist) == list: 

297 hist.append(val) 

298 elif isinstance(hist, np.ndarray): 

299 try: 

300 hist[t_ind] = val 

301 except Exception as e: 

302 obj_str = "Error logging obj "+obj.__class__.__name__+": " 

303 if t_ind >= len(hist): 

304 raise Exception(obj_str + "Time beyond range of model" + 

305 "history--check staged execution " + 

306 "and simulation time settings" + 

307 " (end condition, mdl.sp.end_time)") from e 

308 elif not np.can_cast(type(val), type(hist[t_ind])): 

309 raise Exception(obj_str + str(att)+" changed type: " + 

310 str(type(hist[t_ind])) + " to " + 

311 str(type(val)) + " at t_ind=" + 

312 str(t_ind)) from e 

313 else: 

314 raise Exception(obj_str + "Value too large to represent: " 

315 + att + "=" + str(val)) from e

317 def cut(self, end_ind=None, start_ind=None, newcopy=False): 

318 """ 

319 Cut the history to a given index. 

320 

321 Parameters 

322 ---------- 

323 end_ind : int, optional 

324 the index of the array that you want to cut the history upto. Default is None. 

325 start_ind: int, optional 

326 the index of the array that you want to start cutting the history from.  

327 The default is None, which starts from the 0th index. 

328 newcopy: bool, optional 

329 Tells whether to creat a new history variable with the information what was cut or 

330 cut the original history variable. Default is False. 

331 

332 Examples 

333 -------- 

334 >>> hist = History({'a':[2,3,4,5], 'b':[5,4,3,2,1,0], 'time': [0,1,2,3,4,5]}) 

335 >>> cut_hist = hist.cut(3) 

336 >>> cut_hist 

337 a: array(4) 

338 b: array(4) 

339 time: array(4) 

340 

341 >>> cut_hist.a 

342 [2, 3, 4, 5] 

343 >>> cut_hist.b 

344 [5, 4, 3, 2] 

345 >>> cut_hist.time 

346 [0, 1, 2, 3] 

347 """ 

348 if newcopy: 

349 hist = self.copy() 

350 else: 

351 hist = self 

352 for name, att in hist.items(): 

353 if isinstance(att, History): 

354 hist[name] = hist[name].cut(end_ind, start_ind, newcopy=False) 

355 else: 

356 try: 

357 if end_ind is None: 

358 hist[name] = att[start_ind:] 

359 elif start_ind is None: 

360 hist[name] = att[:end_ind+1] 

361 else: 

362 hist[name] = att[start_ind:end_ind+1] 

363 except TypeError as e: 

364 raise Exception("Invalid history for name " + name + 

365 " in history "+str(hist.data)) from e 

366 return hist

382 def get_fault_time(self, metric="earliest"): 

383 """ 

384 Get the time a fault is present in the system. 

385 

386 Parameters 

387 ---------- 

388 metric : 'earliest','latest','total', optional 

389 Earliest, latest, or total time fault(s) are present. 

390 The default is "earliest". 

391 

392 Returns 

393 ------- 

394 int 

395 index in the history when the fault is present 

396 """ 

397 flatdict = self.flatten() 

398 all_faults_hist = np.sum([v for k, v in flatdict.items() if 'faults' in k], 0) 

399 if metric == 'earliest': 

400 return np.where(all_faults_hist >= 1) 

401 elif metric == 'latest': 

402 return np.where(np.flip(all_faults_hist) >= 1) 

403 elif metric == 'total': 

404 return np.sum(all_faults_hist >= 1)

421 def get_degraded_hist(self, *attrs, nomhist={}, operator=np.any, difftype='bool', 

422 withtime=True, withtotal=True): 

423 """ 

424 Get history of times when the attributes *attrs deviate from nominal values. 

425 

426 Parameters 

427 ---------- 

428 *attrs : names of attributes 

429 Names to check (e.g., `flow_1`, `fxn_2`) 

430 nomhist : History, optional 

431 Nominal history to compare against 

432 (otherwise uses internal nomhist, if available) 

433 operator : function 

434 Method of combining multiple degraded values. The default is np.prod 

435 difftype : 'bool'/'diff'/float 

436 Way to calculate the difference: 

437 

438 - for 'bool', it is calculated as an equality nom == faulty 

439 - for 'diff', it is calculated as a difference nom - faulty 

440 - if a float, is provided, it is calculated as nom - fault > diff 

441 withtime : bool 

442 Whether to include time in the dict. Default is True. 

443 withtotal : bool 

444 Whether to include a total in the dict. Default is True. 

445 

446 Returns 

447 ------- 

448 deghist : History 

449 History of degraded attributes 

450 """ 

451 if not attrs: 

452 attrs = self.keys() 

453 

454 nomhist, faulthist = self._prep_nom_faulty(nomhist) 

455 deghist = History() 

456 for att in attrs: 

457 att_diff = [diff(nomhist[k], v, difftype) 

458 for k, v in faulthist.items() 

459 if att in k] 

460 if att_diff: 

461 deghist[att] = operator(att_diff, 0) 

462 if withtotal: 

463 deghist['total'] = len(deghist.values()) - np.sum([*deghist.values()], axis=0) 

464 if withtime: 

465 deghist['time'] = nomhist['time'] 

466 return deghist

492 def get_faulty_hist(self, *attrs, withtime=True, withtotal=True, operator=np.any): 

493 """ 

494 Get the times when the attributes *attrs have faults present. 

495 

496 Parameters 

497 ---------- 

498 *attrs : names of attributes 

499 Names to check (e.g., `fxn_1`, `fxn_2`) 

500 withtime : bool 

501 Whether to include time in the dict. Default is True. 

502 withtotal : bool 

503 Whether to include a total in the dict. Default is True. 

504 operator : function 

505 Method of combining multiple degraded values. The default is np.any 

506 

507 Returns 

508 ------- 

509 has_faults_hist : History 

510 History of attrs being faulty/not faulty 

511 """ 

512 faulthist = self._prep_faulty() 

513 faults_hist = self.get_faults_hist(*attrs) 

514 has_faults_hist = History() 

515 for att in attrs: 

516 if faults_hist[att]: 

517 has_faults_hist[att] = operator([*faults_hist[att].values()], 0) 

518 if withtotal and has_faults_hist: 

519 has_faults_hist['total'] = np.sum([*has_faults_hist.values()], axis=0) 

520 elif withtotal: 

521 has_faults_hist['total'] = 0 * faulthist['time'] 

522 if withtime: 

523 has_faults_hist['time'] = faulthist['time'] 

524 return has_faults_hist

552 def get_fault_degradation_summary(self, *attrs): 

553 """ 

554 Create a Result with values for the *attrs that are faulty/degraded. 

555 

556 Parameters 

557 ---------- 

558 *attrs : str 

559 Attribute(s) to check. 

560 

561 Returns 

562 ------- 

563 Result 

564 Result dict with structure {'degraded':['degattrname'], 

565 'faulty':['faultyattrname']]} 

566 """ 

567 faulty_hist = self.get_faulty_hist(*attrs, withtotal=False, withtime=False) 

568 faulty = [k for k, v in faulty_hist.items() if np.any(v)] 

569 deg_hist = self.get_degraded_hist(*attrs, withtotal=False, withtime=False) 

570 degraded = [k for k, v in deg_hist.items() if not np.all(v)] 

571 return Result(faulty=faulty, degraded=degraded)

650 """ 

651 simhists, plot_values, grouphists, indiv_kwargs = prep_hists(self, 

652 plot_values, 

653 comp_groups, 

654 indiv_kwargs) 

655 fig, axs, cols, rows, subplot_titles = multiplot_helper(cols, *plot_values, 

656 figsize=figsize, 

657 titles=titles) 

658 

659 for i, value in enumerate(plot_values): 

660 ax = axs[i] 

661 ax.grid() 

662 if i >= (rows-1)*cols and xlabel: 

663 xlab = xlabel 

664 else: 

665 xlab = ' ' 

666 ylab = ylabels.get(value, '') 

667 for group, hists in grouphists.items(): 

668 loc_kwargs = {**kwargs, 'label': group, 'xlabel': xlab, 'ylabel': ylab, 

669 'title': subplot_titles[value], 

670 **indiv_kwargs.get(group, {})} 

671 try: 

672 if aggregation == 'individual': 

673 hists.plot_individual_line(value, fig, ax, **loc_kwargs) 

674 elif aggregation == 'mean_std': 

675 hists.plot_mean_std_line(value, fig, ax, **loc_kwargs) 

676 elif aggregation == 'mean_ci': 

677 hists.plot_mean_ci_line(value, fig, ax, max_ind=max_ind, 

678 **loc_kwargs) 

679 elif aggregation == 'mean_bound': 

680 hists.plot_mean_bound_line(value, fig, ax, **loc_kwargs) 

681 elif aggregation == 'percentile': 

682 hists.plot_percentile_line(value, fig, ax, **loc_kwargs) 

683 else: 

684 raise Exception("Invalid aggregation option: "+aggregation) 

685 except Exception as e: 

686 raise Exception("Error at plot_value " + str(value) 

687 + " and group: " + str(group)) from e 

688 if len(value) > 1 and value[1] in phases: 

689 phase_overlay(ax, phases[value[1]], value[1]) 

690 if type(time_slice) == int: 

691 ax.axvline(x=time_slice, color='k', label=time_slice_label) 

692 else: 

693 for ts in time_slice: 

694 ax.axvline(x=ts, color='k', label=time_slice_label) 

695 set_empty_multiplots(axs, len(plot_values), cols, xlab_ang=0.0, grid=True) 

696 multiplot_legend_title(grouphists, axs, ax, legend_loc, title, 

697 v_padding, h_padding, title_padding, legend_title) 

698 return fig, axs

719 def get_mean_std_errhist(self, value): 

720 """ 

721 Get aggregated err_hist of means surrounded by std deviation. 

722 

723 Parameters 

724 ---------- 

725 value : str 

726 Value to get mean and bounds of. 

727 

728 Returns 

729 ------- 

730 err_hist : History

739 hist['low'] = hist['stat']-std_dev/2 

740 return hist

747 def get_mean_ci_errhist(self, value, ci=0.95, max_ind='max'): 

748 """ 

749 Get aggregated err_hist of means surrounded by confidence intervals. 

750 

751 Parameters 

752 ---------- 

753 value : str 

754 Value to get mean and bounds of. 

755 ci : float 

756 Fraction for confidence interval. Default is 0.95. 

757 max_ind : str/int 

758 Max index of time to clip to. Default is 'max'. 

759 

760 Returns 

761 ------- 

762 err_hist : History 

763 hist of line, low, high values. Has the form:: 

764 {'time': times, 'stat': stat_values, 'low': low_values, 'high': high_values} 

765 """ 

766 hist = History() 

767 hist['time'] = self.get_metric('time', axis=0) 

768 hist['stat'] = self.get_metric(value, np.mean, axis=0) 

769 if max_ind == 'max': 

770 max_ind = min([len(h) for h in self.values()]) 

771 vals = [[hist[t] for hist in self.get_values(value).values()] 

772 for t in range(max_ind)] 

773 boot_stats = np.array([bootstrap_confidence_interval(val, return_anyway=True, 

774 confidence_level=ci) 

775 for val in vals]).transpose() 

776 hist['high'] = boot_stats[1] 

777 hist['low'] = boot_stats[2] 

778 return hist

860 def plot_metric_dist(self, times, *plot_values, **kwargs): 

861 """ 

862 Plot the distribution of values at defined time(s) over a number of scenarios. 

863 

864 Parameters 

865 ---------- 

866 times : list/int 

867 List of times (or single time) to key the model history from. 

868 If more than one time is provided, it takes the place of comp_groups. 

869 *plot_values : strs 

870 names of values to pull from the history (e.g., 'fxns.move_water.s.flow') 

871 Can also be specified as a dict (e.g., {'fxns':'move_water'}) to get all keys 

872 from a given fxn/flow/mode/etc. 

873 **kwargs : kwargs 

874 keyword arguments to Result.plot_metric_dist 

875 """ 

876 flat_mdlhists = self.nest(levels=1) 

877 if type(times) in [int, float]: 

878 times = [times] 

879 if len(times) == 1 and kwargs.get('comp_groups', False): 

880 time_classes = Result({scen: Result(flat_hist.get_slice(times[0])) 

881 for scen, flat_hist in flat_mdlhists.items()}) 

882 comp_groups = kwargs.pop('comp_groups') 

883 elif kwargs.get('comp_groups', False): 

884 raise Exception("Cannot compare times and comp_groups at the same time") 

885 else: 

886 time_classes = Result({str(t)+'_'+scen: Result(flat_hist.get_slice(t)) 

887 for scen, flat_hist in flat_mdlhists.items() 

888 for t in times}) 

889 comp_groups = {str(t): {str(t)+'_'+scen for scen in flat_mdlhists} 

890 for t in times} 

891 res_to_plot = time_classes.flatten() 

892 fig, axs = res_to_plot.plot_metric_dist(*plot_values, 

893 comp_groups=comp_groups, **kwargs) 

894 return fig, axs

901 """ 

902 Plot trajectories from the environment in 2d or 3d space. 

903 

904 Parameters 

905 ---------- 

906 *plot_values : str 

907 Plot values corresponding to the x/y/z values (e.g, 'position.s.x') 

908 comp_groups : dict, optional

910 ::

918 Dict of kwargs to use to differentiate each comparison group. 

919 Has structure:: 

920 {comp1: kwargs1, comp2: kwargs2} 

921

932 each column and 2 for each row. 

933 time_groups : list, optional

935 at each point in the trajectory. The default is []. 

936 time_ticks : float, optional

938 time_fontsize : int, optional 

939 Fontsize for time-ticks. The default is 8. 

940 xlim : tuple, optional 

941 Limits on the x-axis. The default is (). 

942 ylim : tuple, optional 

943 Limits on the y-axis. The default is (). 

944 zlim : tuple, optional 

945 Limits on the z-axis. The default is (). 

946 legend : bool, optional 

947 Whether to show a legend. The default is True. 

948 title : str, optional 

949 Title to add. Default is '' (no title). 

950 fig : matplotlib.figure, optional 

951 Existing Figure. The default is None. 

952 ax : matplotlib.axis, optional 

953 Existing axis. The default is None. 

954 **kwargs : kwargs 

955 kwargs to ax.plot to use over all plots. 

956 

957 Returns 

958 ------- 

959 fig : figure 

960 Matplotlib figure object 

961 ax : axis 

962 Corresponding matplotlib axis

978 time_ticks=time_ticks, time_fontsize=time_fontsize)

993 ax.set_title(title)
