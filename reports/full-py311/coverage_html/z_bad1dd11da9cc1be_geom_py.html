33class GeomArchitecture(Architecture): 

34 """ 

35 Agglomeration of multiple geoms/shapes. 

36 

37 Architecture is defined using add_shape method in user-defined init_shapes method. 

38 

39 Examples 

40 -------- 

41 for an architecture with the geoms already defined: 

42 

43 >>> class ExGeomArch(GeomArchitecture): 

44 ... def init_architecture(self): 

45 ... self.add_point("ex_point", ExPoint) 

46 ... self.add_line("ex_line", ExLine) 

47 ... self.add_poly("ex_poly", ExPoly) 

48 

49 This can then be used in containing classes (e.g., environments) that need multiple 

50 geoms. We can then access the individual geoms in the geoms dict, e.g.,: 

51 

52 >>> ega = ExGeomArch() 

53 >>> ega.geoms()['ex_point'].s 

54 ExGeomState(occupied=False) 

55 >>> ega.h 

56 points.ex_point.s.occupied: array(101) 

57 lines.ex_line.s.occupied: array(101) 

58 polys.ex_poly.s.occupied: array(101) 

59 >>> ega.return_mutables() 

60 ((False,), (False,), (False,), (-0.1, 0, 0.0, 1)) 

61 """

81 def check_geom_class(self, gclass, baseclass): 

82 """Check that geom class/object inherits from given base class.""" 

83 try: 

84 if not issubclass(gclass, baseclass): 

85 raise Exception("gclass "+gclass+" not a "+baseclass.__name__) 

86 except TypeError: 

87 if not isinstance(gclass, baseclass): 

88 raise Exception("gclass "+gclass+" not a "+baseclass.__name__)

138 def all_at(self, *pt): 

139 """ 

140 Find all geoms (and buffers) a given is at. 

141 

142 Parameters 

143 ---------- 

144 *pt : x,y 

145 x, y, z location to check. 

146 

147 Returns 

148 ------- 

149 all_at : dict 

150 Names of geoms where the point is at (and their properties) 

151 

152 Examples 

153 -------- 

154 >>> exga = ExGeomArch() 

155 >>> exga.all_at(1.0, 1.0) 

156 {'ex_point': ['shape', 'on'], 'ex_line': ['shape', 'on'], 'ex_poly': ['shape']} 

157 >>> exga.all_at(0.0, 0.0) 

158 {'ex_line': ['shape', 'on'], 'ex_poly': ['shape']} 

159 >>> exga.all_at(0.4, 0.3) 

160 {'ex_point': ['on'], 'ex_line': ['on']} 

161 """ 

162 all_at = {} 

163 for geomname, geom in self.geoms().items(): 

164 at_geom = geom.all_at(*pt) 

165 if at_geom: 

166 all_at[geomname] = at_geom 

167 return all_at

169 def show(self, geoms={'all': {}}, fig=None, ax=None, figsize=(4, 4), z=False, 

170 **kwargs): 

171 """ 

172 Show the shapes of a GeomArchitecture all on one plot. 

173 

174 Parameters 

175 ---------- 

176 geoms : dict, optional 

177 Individual shapes to plot and their corresponding kwargs. 

178 The default is {'all': {}}. 

179 fig : matplotlib.figure, optional 

180 Existing Figure. The default is None. 

181 ax : matplotlib.axis, optional 

182 Existing axis. The default is None. 

183 figsize : tuple, optional 

184 Size for figure (if instantiating). The default is (4, 4). 

185 z : bool/number, optional 

186 If plotting on a 3d axis, set z to a number which will be the z-level. 

187 The default is False. 

188 **kwargs : kwargs 

189 Overall kwargs to show.geom for all geoms. 

190 

191 Returns 

192 ------- 

193 fig : figure 

194 Matplotlib figure object 

195 ax : axis 

196 Corresponding matplotlib axis 

197 """ 

198 if not ax: 

199 fig, ax = setup_plot(z=z, figsize=figsize) 

200 if 'all' in geoms: 

201 geoms = {g: {'shapes': 'all'} for g in self.geoms()} 

202 

203 for geomname, geom_kwargs in geoms.items(): 

204 local_kwargs = {**kwargs, 'geomlabel': geomname, **geom_kwargs} 

205 fig, ax = self.geoms()[geomname].show(ax=ax, fig=fig, z=z, **local_kwargs) 

206 return fig, ax
