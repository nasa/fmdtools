21 

22import unittest 

23import numpy as np 

24 

25 

26class define_Tests(unittest.TestCase): 

27 def test_pdf_translation_options(self): 

28 """ 

29 Test for getting the probability of a pdf using get_pdf_for_rand. 

30 Ensures that arguments are passed to scipy and a probability is returned 

31 

32 Also spot-tests values for certain distributions 

33 (integers, random, normal, choice) 

34 """ 

35 rands = {'integers': (4,), 

36 'random': (None,), 

37 'shuffle': ([1, 2],), 

38 'permutation': ([1, 2],), 

39 'permuted': ([1, 2],), 

40 'choice': ([1, 2, 3],)} 

41 same_funcs = {'beta': (1, 2), 

42 'dirichlet': ([0.5],), 

43 'f': (1, 2), 

44 'gamma': (1,), 

45 'laplace': (0, 1), 

46 'logistic': (0, 1), 

47 'multivariate_normal': (0, 1), 

48 'pareto': (1,), 

49 'uniform': (0, 1), 

50 'wald': (0, 1)} 

51 same_funcs_pmf = {'multinomial': (2, [0.2]), 

52 'poisson': (1,), 

53 'zipf': (2,)} 

54 different_funcs_pmf = {'binomial': (2, 0.5), 

55 'geometric': (0.5,), 

56 'logseries': (0.5,), 

57 'multivariate_hypergeometric': (8, 2), 

58 'negative_binomial': (0.5, 2)} 

59 different_funcs = {'chisquare': (3,), 

60 'gumbel': (1, 1), 

61 'noncentral_chisquare': (3, 0.1), 

62 'noncentral_f': (2, 3, 0.1), 

63 'normal': (1, 1), 

64 'power': (1.0,), 

65 'standard_cauchy': (), 

66 'standard_gamma': (0.5,), 

67 'standard_normal': (), 

68 'weibull': (0.5,)} 

69 randnames = {**rands, **same_funcs, **same_funcs_pmf, **different_funcs} 

70 

71 x = 1 

72 expected_values = {'integers': 0.25, 

73 'random': 1, 

74 'shuffle': 0.5, 

75 'permuted': 0.5, 

76 'choice': 1/3, 

77 'normal': 0.398942, 

78 'pareto': 1.0, 

79 'poisson': 0.368, 

80 'power': 1.0, 

81 'standard_normal': 0.241971} 

82 

83 for randname in randnames: 

84 p_d = get_pdf_for_rand(x, randname, randnames[randname]) 

85 self.assertLessEqual(p_d[0], 1.0) # checks to see that probability is 0<x<1 

86 # note that some densities may be higher than this under some values 

87 # this is mainly a check 

88 self.assertGreaterEqual(p_d[0], 0.0) 

89 if randname in expected_values: # spot tests for common distributions 

90 self.assertAlmostEqual(p_d[0], expected_values[randname], 3) 

91 self.assertIsInstance(p_d, np.ndarray) 

92 

93 

94if __name__ == '__main__': 

95 unittest.main()
