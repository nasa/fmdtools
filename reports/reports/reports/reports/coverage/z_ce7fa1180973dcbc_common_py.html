41def get_sub_include(att, to_include): 

42 """Determine attributes of att to include based on provided dict/str/list/set.""" 

43 if type(to_include) in [list, set, tuple, str]: 

44 if att in to_include: 

45 new_to_include = 'default' 

46 elif isinstance(to_include, str) and to_include == 'all': 

47 new_to_include = 'all' 

48 elif isinstance(to_include, str) and to_include == 'default': 

49 new_to_include = 'default'

51 new_to_include = False

59def to_include_keys(to_include): 

60 """ 

61 Determine dict keys to include from Result given nested to_include dictionary. 

62 

63 Examples 

64 -------- 

65 >>> to_include_keys({"a":{"b": "c"}}) 

66 ('a.b.c',) 

67 >>> ks = to_include_keys({"a":{"b": {"c", "d", "e"}}}) 

68 >>> all([k in ks for k in ('a.b.c', 'a.b.e', 'a.b.d')]) 

69 True 

70 >>> to_include_keys("hi") 

71 ('hi',) 

72 >>> to_include_keys(["a", "b", "c"]) 

73 ('a', 'b', 'c') 

74 """ 

75 if isinstance(to_include, str): 

76 return tuple([to_include]) 

77 elif type(to_include) in [list, set, tuple]: 

78 return tuple([to_i for to_i in to_include]) 

79 elif isinstance(to_include, dict): 

80 keys = [] 

81 for k, v in to_include.items(): 

82 add = to_include_keys(v) 

83 keys.extend([k+'.'+v for v in add]) 

84 return tuple(keys)

87def is_numeric(val): 

88 """ 

89 Check if a given value is a number. 

90 

91 Examples 

92 -------- 

93 >>> is_numeric(1.0) 

94 True 

95 >>> is_numeric("hi") 

96 False 

97 >>> is_numeric(np.array([1.0])[0]) 

98 True 

99 >>> is_numeric(np.array(["hi"])[0]) 

100 False 

101 """ 

102 try: 

103 return np.issubdtype(np.array(val).dtype, np.number) 

104 except TypeError: 

105 return type(val) in [float, bool, int]

108def bootstrap_confidence_interval(data, method=np.mean, return_anyway=False, **kwargs): 

109 """ 

110 Return bootstrap confidence interval (helper for scipy.bootstrap). 

111 

112 Parameters 

113 ---------- 

114 data : list/array/etc 

115 Iterable with the data. May be float (for mean) or indicator (for proportion) 

116 method : method 

117 numpy method to give scipy.bootstrap. 

118 return_anyway: bool 

119 Gives a dummy interval of (stat, stat) if no . Used for plotting 

120 Returns 

121 ---------- 

122 statistic, lower bound, upper bound 

123 """ 

124 from scipy.stats import bootstrap 

125 if 'interval' in kwargs: 

126 kwargs['confidence_level'] = kwargs.pop('interval')*0.01 

127 if data.count(data[0]) != len(data): 

128 bs = bootstrap([data], np.mean, **kwargs) 

129 return method(data), bs.confidence_interval.low, bs.confidence_interval.high 

130 elif return_anyway: 

131 return method(data), method(data), method(data) 

132 else: 

133 raise Exception("All data are the same!")

209def phase_overlay(ax, phasemap, label_phases=True): 

210 """Overlay phasemap information on plot.""" 

211 ymin, ymax = ax.get_ylim() 

212 phaseseps = [i[0] for i in list(phasemap.phases.values())[1:]] 

213 ax.vlines(phaseseps, ymin, ymax, colors='gray', linestyles='dashed') 

214 if label_phases: 

215 for phase in phasemap.phases: 

216 if phasemap.modephases: 

217 phasetext = [m for m, p in phasemap.modephases.items() if phase in p][0] 

218 else: 

219 phasetext = phase 

220 bbox_props = dict(boxstyle="round,pad=0.3", fc="white", lw=0, alpha=0.5) 

221 ax.text(np.average(phasemap.phases[phase]), 

222 (ymin+ymax)/2, phasetext, ha='center', bbox=bbox_props)

225def plot_err_hist(err_hist, ax=None, fig=None, figsize=(6, 4), boundtype='fill', 

226 boundcolor='gray', boundlinestyle='--', fillalpha=0.3, 

227 xlabel='time', ylabel='', title='', **kwargs): 

228 """ 

229 Plot a line with a given range of uncertainty around it. 

230 

231 Parameters 

232 ---------- 

233 err_hist : History 

234 hist of line, low, high values. Has the form :: 

235 {'time': times, 'stat': stat_values, 'low': low_values, 'high': high_values} 

236 ax : mpl axis (optional) 

237 axis to plot the line on 

238 fig : mpl figure (optional) 

239 figure to plot line on 

240 figsize : tuple 

241 figure size (optional) 

242 boundtype : 'fill' or 'line' 

243 Whether the bounds should be marked with lines or a fill 

244 boundcolor : str, optional 

245 Color for bound fill The default is 'gray'. 

246 boundlinestyle : str, optional 

247 linestyle for bound lines (if any). The default is '--'. 

248 fillalpha : float, optional 

249 Alpha for fill. The default is 0.3. 

250 **kwargs : kwargs 

251 kwargs for the line 

252 

253 Returns 

254 ------- 

255 fig : mpl figure 

256 ax :mpl, axis 

257 """ 

258 fig, ax = setup_plot(fig, ax, figsize) 

259 ax.plot(err_hist['stat'], **kwargs) 

260 if boundtype == 'fill': 

261 col = ax.lines[-1].get_color() 

262 ax.fill_between(err_hist['time'], err_hist['low'], err_hist['high'], 

263 alpha=fillalpha, color=col) 

264 if 'med_high' in err_hist and 'med_low' in err_hist: 

265 ax.fill_between(err_hist['time'], err_hist['med_low'], err_hist['med_high'], 

266 alpha=fillalpha, color=col) 

267 elif boundtype == 'line': 

268 plot_err_lines(err_hist['time'], err_hist['low'], err_hist['high'], ax=ax, 

269 fig=fig, color=boundcolor, linestyle=boundlinestyle) 

270 if 'med_high' in err_hist and 'med_low' in err_hist: 

271 plot_err_lines(err_hist['time'], err_hist['med_low'], err_hist['med_high'], 

272 ax=ax, fig=fig, color=boundcolor, linestyle=boundlinestyle) 

273 else: 

274 raise Exception("Invalid bound type: "+boundtype) 

275 add_title_xylabs(ax, title=title, xlabel=xlabel, ylabel=ylabel) 

276 ax.set_xlim(err_hist['time'][0], err_hist['time'][-1]) 

277 return fig, ax

338def multiplot_helper(cols, *plot_values, figsize='default', titles={}, sharex=True, 

339 sharey=False, fig=None, axs=None): 

340 """Create multiple plot axes for plotting.""" 

341 num_plots = len(plot_values) 

342 if num_plots == 1: 

343 cols = 1 

344 rows = int(np.ceil(num_plots/cols)) 

345 if not fig or not axs: 

346 if figsize == 'default': 

347 figsize = (cols*3, 2*rows) 

348 if not fig: 

349 fig, axs = plt.subplots(rows, cols, 

350 sharex=sharex, sharey=sharey, figsize=figsize) 

351 if axs is None: 

352 if len(fig.axes) != num_plots: 

353 fig.clf() 

354 axs = fig.subplots(rows, cols, sharex=sharex, sharey=sharey) 

355 else: 

356 axs = fig.axes 

357 

358 if isinstance(axs, np.ndarray): 

359 axs = axs.flatten() 

360 elif not isinstance(axs, list): 

361 axs = [axs] 

362 

363 subplot_titles = {plot_value: plot_value for plot_value in plot_values} 

364 subplot_titles.update(titles) 

365 return fig, axs, cols, rows, subplot_titles

368def set_empty_multiplots(axs, num_plots, cols, xlab_ang=-90, grid=False, 

369 set_above=True): 

370 """Align empty axes with the rest of the multiplot.""" 

371 num_empty = len(axs) - num_plots 

372 starting_ax = len(axs) - num_empty 

373 for i, ax in enumerate(axs): 

374 if i >= starting_ax: 

375 # clear empty box 

376 ax.set_frame_on(False) 

377 ax.get_xaxis().set_visible(False) 

378 ax.get_yaxis().set_visible(False) 

379 # set x tick labels for axis above 

380 if set_above: 

381 ax_above = axs[i-cols] 

382 ax_set = axs[num_plots-1] 

383 ax_above.tick_params(axis='x', rotation=xlab_ang, reset=True) 

384 ax_above.set_xlabel(ax_set.get_xlabel()) 

385 labels = [t.get_text() for t in ax_set.get_xticklabels()] 

386 ax_above.set_xticks(ax_set.get_xticks(), labels=labels) 

387 # turn back on the reset grid 

388 if grid: 

389 ax_above.grid(axis='x')

394 title_padding=0.0, legend_title=None):

413def consolidate_legend(ax, loc='upper left', bbox_to_anchor=(1.05, 1), 

414 add_handles=[], remove_empty=True, color='', **kwargs): 

415 """ 

416 Create a single legend for plots where multiple groups are being compared. 

417 

418 Parameters 

419 ---------- 

420 ax : matplotlib axis 

421 Axis object to mark on 

422 loc : str 

423 Legend location (see matplotlib.legend). Default is 'upper left'. 

424 bbox_to_anchor : tuple 

425 Anchoring bounding box point (see matplotlib.legend). Default is (1.05, 1). 

426 add_handles : list 

427 Labelled plot handles to add to the legend. Default is []. 

428 remove_empty : bool 

429 If add_handles is used, this toggles whether to add unlabeled entries to the 

430 legend. Default is False. 

431 """ 

432 # get handles/labels and any existing legend 

433 hands, labs = ax.get_legend_handles_labels() 

434 old_legend = ax.get_legend() 

435 

436 # if there is an old legend (which may not have these handles), 

437 # update it with new handles/labels 

438 if old_legend: 

439 old_hands = old_legend.legend_handles 

440 ax.get_legend().remove() 

441 hands = old_hands + hands 

442 

443 # if there are labels to add (add_handles argument), add them here 

444 handles, labels = [], [] 

445 for handle in hands + add_handles: 

446 lab = handle.get_label() 

447 if not (not lab and remove_empty): 

448 handles.append(handle) 

449 labels.append(lab) 

450 by_label = dict(zip(labels, handles)) 

451 

452 # generate legend with consolidated labels/handles 

453 ax.legend(by_label.values(), by_label.keys(), 

454 bbox_to_anchor=bbox_to_anchor, loc=loc, **kwargs)

481def suite_for_plots(testclass, plottests=False): 

482 """ 

483 Qualitative testing suite with or without plots in unittest. 

484 

485 Plot tests should have "plot" in the title of their method, this enables this 

486 function tofilter them out (or include them). 

487 

488 Parameters 

489 ---------- 

490 testclass : unittest.TestCase 

491 Test-case to create the suite for. 

492 plottests : bool/list, optional 

493 Whether to show the plot tests (True) or the non-plot tests (False). If a 

494 list is provided, only tests provided in the list will be run. 

495 

496 Returns 

497 ------- 

498 suite : unittest.TestSuite 

499 Test Suite to run with unittest.TextTestRunner() using runner.run 

500 (e.g., runner = unittest.TextTestRunner(); 

501 runner.run(suite_for_plots(UnitTests, plottests=False))) 

502 """ 

503 import unittest 

504 suite = unittest.TestSuite() 

505 if not plottests: 

506 tests = [func for func in dir(testclass) 

507 if (func.startswith("test") and not ('plot' in func))] 

508 elif isinstance(plottests, list): 

509 tests = [func for func in dir(testclass) 

510 if (func.startswith("test") and func in plottests)] 

511 else: 

512 tests = [func for func in dir(testclass) 

513 if (func.startswith("test") and 'plot' in func)] 

514 for test in tests: 

515 suite.addTest(testclass(test)) 

516 return suite
