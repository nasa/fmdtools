56class LineArchParam(Parameter): 

57 """ 

58 Line Architecture defined by parameter 'archtype'. 

59 

60 Archtype has options: 

61 - 'quad': quadrotor architecture 

62 - 'hex': hexarotor architecture 

63 - 'oct': octorotor architecture 

64 Which in turn change the following fields... 

65 

66 Fields 

67 ------- 

68 components: tuple 

69 Set of component names for the lines named using the convention for the str: 

70 0: l/r - left or right 

71 1: f/r - front or rear 

72 3: /2: - if 2, this is a secondary rotor in a similar location. 

73 lr_dict: dict 

74 Left/right dictionary. Has structure {"l": {<left components>}, ...} 

75 fr_dict: dict 

76 Front/rear dictionary. Has structure {'f':{<front components>}, ...} 

77 opposite: 

78 Component on the opposite side of a given component. Used for reconfiguration. 

79 

80 Examples 

81 -------- 

82 >>> LineArchParam() 

83 LineArchParam(archtype='quad', components=('lf', 'lr', 'rf', 'rr'), lr_dict={'l': ('lf', 'lr'), 'r': ('rf', 'rr')}, fr_dict={'f': ('lf', 'rf'), 'r': ('lr', 'rr')}, opposite={'rf': 'lr', 'rr': 'lf', 'lr': 'rf', 'lf': 'rr'}) 

84 >>> LineArchParam(archtype='hex') 

85 LineArchParam(archtype='hex', components=('lf', 'lr', 'rf', 'rr', 'f', 'r'), lr_dict={'l': ('lf', 'lr'), 'r': ('rf', 'rr')}, fr_dict={'f': ('lf', 'rf', 'f'), 'r': ('lr', 'rr', 'r')}, opposite={'rf': 'lr', 'rr': 'lf', 'f': 'r', 'lr': 'rf', 'lf': 'rr', 'r': 'f'}) 

86 >>> LineArchParam(archtype='oct') 

87 LineArchParam(archtype='oct', components=('lf', 'lr', 'rf', 'rr', 'lf2', 'lr2', 'rf2', 'rr2'), lr_dict={'l': ('lf', 'lr', 'lf2', 'lr2'), 'r': ('rf', 'rr', 'rf2', 'rr2')}, fr_dict={'f': ('lf', 'rf', 'lf2', 'rf2'), 'r': ('lr', 'rr', 'lr2', 'rr2')}, opposite={'rf': 'lr', 'rr': 'lf', 'rf2': 'lr2', 'rr2': 'lf2', 'lr': 'rf', 'lf': 'rr', 'lr2': 'rf2', 'lf2': 'rr2'}) 

88 """ 

89 

90 archtype: str = 'quad' 

91 components: tuple = () 

92 lr_dict: dict = dict() 

93 fr_dict: dict = dict() 

94 opposite: dict = dict() 

95 

96 def __init__(self, *args, **kwargs): 

97 archtype = self.get_true_field('archtype', *args, **kwargs) 

98 if archtype == 'quad': 

99 components = ('lf', 'lr', 'rf', 'rr') 

100 lr_dict = {'l': ('lf', 'lr'), 'r': ('rf', 'rr')} 

101 fr_dict = {'f': ('lf', 'rf'), 'r': ('lr', 'rr')} 

102 opposite = {'rf': 'lr', 'rr': 'lf'} 

103 elif archtype == 'hex': 

104 components = ('lf', 'lr', 'rf', 'rr', 'f', 'r') 

105 lr_dict = {'l': ('lf', 'lr'), 'r': ('rf', 'rr')} 

106 fr_dict = {'f': ('lf', 'rf', 'f'), 'r': ('lr', 'rr', 'r')} 

107 opposite = {'rf': 'lr', 'rr': 'lf', 'f': 'r'} 

108 elif archtype == 'oct': 

109 components = ('lf', 'lr', 'rf', 'rr', 'lf2', 'lr2', 'rf2', 'rr2') 

110 lr_dict = {'l': ('lf', 'lr', 'lf2', 'lr2'), 'r': ('rf', 'rr', 'rf2', 'rr2')} 

111 fr_dict = {'f': ('lf', 'rf', 'lf2', 'rf2'), 'r': ('lr', 'rr', 'lr2', 'rr2')} 

112 opposite = {'rf': 'lr', 'rr': 'lf', 'rf2': 'lr2', 'rr2': 'lf2'} 

113 else: 

114 raise Exception("Invalid arch type") 

115 opposite.update({v: k for k, v in opposite.items()}) 

116 args = self.get_true_fields(*args, archtype=archtype, components=components, 

117 lr_dict=lr_dict, fr_dict=fr_dict, opposite=opposite) 

118 super().__init__(*args, strict_immutability=False)

135class AffectDOFArch(ComponentArchitecture): 

136 container_p = LineArchParam 

137 container_s = LineArchState 

138 

139 def init_architecture(self, **kwargs): 

140 # add lines 

141 for compname in self.p.components: 

142 self.add_comp(compname, Line) 

143 # add state configuration - relative throttle for each line 

144 if self.p.archtype == 'quad': 

145 self.s.forward.update({'rf': 0.5, 'lf': 0.5, 'lr': -0.5, 'rr': -0.5}) 

146 elif self.p.archtype == 'hex': 

147 self.s.forward.update({'rf': 0.5, 'lf': 0.5, 'lr': -0.5, 

148 'rr': -0.5, 'r': -0.75, 'f': 0.75}) 

149 elif self.p.archtype == 'oct': 

150 self.s.forward.update({'rf': 0.5, 'lf': 0.5, 'lr': -0.5, 'rr': -0.5, 

151 'rf2': 0.5, 'lf2': 0.5, 'lr2': -0.5, 'rr2': -0.5}) 

152 self.s.upward = {c: 1.0 for c in self.p.components}

155class AffectDOF(AffectDOFDynamic): 

156 """Rotor locomotion (multi-component extension)."""

158 __slots__ = () 

159 container_s = OverallAffectDOFState

162 def static_behavior(self, time): 

163 """Rotor dynamic behavior with architecture-base recovery.""" 

164 if self.ca.p.opposite: 

165 self.reconfig_faults() 

166 self.calc_pwr() 

167 

168 def reconfig_faults(self): 

169 """Corrects for individual line faultmodes by turning off the opposite rotor 

170 and upping the throttle (amp_factor)""" 

171 for fault in self.m.faults: 

172 if fault in self.ca.faultmodes: 

173 comp = self.ca.faultmodes[fault] 

174 opp = self.ca.p.opposite[comp] 

175 if self.ca.s.forward[comp] != 0.0: 

176 self.ca.s.forward[comp] = 0.0 

177 self.ca.s.upward[comp] = 0.0 

178 if self.ca.s.forward[opp] != 0.0: 

179 self.ca.s.forward[opp] = 0.0 

180 self.ca.s.upward[opp] = 0.0 

181 tot_comps = len(self.ca.comps) 

182 empty_comps = len([c for c in self.ca.s.forward if self.ca.s.forward[c] == 0.0]) 

183 try: 

184 self.s.amp_factor = tot_comps / (tot_comps - empty_comps) 

185 except ZeroDivisionError: 

186 self.s.amp_factor = 1.0 

187 

188 def calc_pwr(self): 

189 """ 

190 Calculates overall power and stability based on individual rotor output. 

191 

192 e.g., :: 

193 >>> a = AffectDOF() 

194 >>> a.dofs.s.put(z=100.0) 

195 >>> a.ctl_in.s.put(forward=0.0, upward=1.0) 

196 >>> a.calc_pwr() 

197 >>> a.dofs.s 

198 DOFstate(vertvel=1.0, planvel=1.0, planpwr=-0.0, uppwr=1.0, x=0.0, y=0.0, z=100.0) 

199 >>> a.ctl_in.s.put(forward=1.0, upward=1.0) 

200 >>> a.calc_pwr() 

201 >>> a.dofs.s 

202 DOFstate(vertvel=1.0, planvel=1.0, planpwr=1.0, uppwr=1.0, x=0.0, y=0.0, z=100.0) 

203 """ 

204 air, ee_in = {}, {} 

205 # injects faults into lines 

206 for linname, lin in self.ca.comps.items(): 

207 a, ee = lin.behavior(self.ee_in.s.effort, 

208 self.ctl_in, 

209 self.ca.s.upward[linname] * self.s.amp_factor, 

210 self.ca.s.forward[linname] * self.s.amp_factor, 

211 self.force.s.support) 

212 air[lin.name] = a 

213 ee_in[lin.name] = ee 

214 

215 if any(value >= 10 for value in ee_in.values()): 

216 self.ee_in.s.rate = 10 

217 elif any(value != 0.0 for value in ee_in.values()): 

218 self.ee_in.s.rate = sum(ee_in.values()) / \ 

219 len(ee_in) # should it really be max? 

220 else: 

221 self.ee_in.s.rate = 0.0 

222 

223 self.s.lrstab = (sum([air[comp] for comp in self.ca.p.lr_dict['l']]) - 

224 sum([air[comp] for comp in self.ca.p.lr_dict['r']]))/len(air) 

225 self.s.frstab = (sum([air[comp] for comp in self.ca.p.fr_dict['r']]) - 

226 sum([air[comp] for comp in self.ca.p.fr_dict['f']]))/len(air) 

227 if abs(self.s.lrstab) >= 0.4 or abs(self.s.frstab) >= 0.75: 

228 self.dofs.s.put(uppwr=0.0, planpwr=0.0) 

229 else: 

230 airs = list(air.values()) 

231 self.dofs.s.uppwr = np.mean(airs) 

232 self.dofs.s.planpwr = -2*self.s.frstab

235class Line(Component, BaseLine): 

236 """Individual version of a line (extends BaseLine in static model).""" 

237 

238 __slots__ = () 

239 container_s = AffectDOFState 

240 container_m = AffectDOFMode 

241 

242 def behavior(self, ee_in, ctlin, u_fact, f_fact, force): 

243 """Calculate air, ee out based on inputs and modes.""" 

244 if force <= 0.0: 

245 self.m.add_fault('mechbreak', 'propbreak') 

246 elif force <= 0.5: 

247 self.m.add_fault('mechfriction') 

248 

249 self.calc_faults() 

250 

251 pwr = ctlin.s.upward * u_fact + ctlin.s.forward * f_fact 

252 airout = m2to1([ee_in, 

253 self.s.e_ti, 

254 pwr, 

255 self.s.ct, 

256 self.s.mt, 

257 self.s.pt]) 

258 ee_in = m2to1([ee_in, self.s.e_to, pwr]) 

259 return airout, ee_in
