33class Function(Block): 

34 """ 

35 Superclass for representing system functions. 

36 

37 Functions are distinguished from other blocks in their ability to contain 

38 architectures, which may be used to hold multiple components or an action sequence 

39 graph within the function. 

40 

41 

42 Additional role types 

43 --------------------- 

44 arch : Architecture 

45 component, action, function architectures at ca, aa, fa, etc. 

46 

47 Examples 

48 -------- 

49 >>> exf = ExampleFunction("exf") 

50 >>> exf 

51 exf ExampleFunction 

52 - ExampleState(x=1.0, y=1.0) 

53 - ExampleMode(mode=standby, faults=set()) 

54 

55 Behavior can be called using __call__ or the user-defined behavior method: 

56 

57 >>> exf("dynamic", time=1.0) 

58 >>> exf 

59 exf ExampleFunction 

60 - ExampleState(x=2.0, y=1.0) 

61 - ExampleMode(mode=standby, faults=set()) 

62 

63 Which can also be used to inject faults: 

64 

65 >>> exf("dynamic", time=2.0, faults=['no_charge']) 

66 >>> exf 

67 exf ExampleFunction 

68 - ExampleState(x=2.0, y=4.0) 

69 - ExampleMode(mode=no_charge, faults={'no_charge'}) 

70 """ 

71 

72 __slots__ = ["ca", "aa", "fa", "args_f", "archs"] 

73 default_track = ["ca", "aa", "fa"]+Block.default_track 

74 roletypes = ['container', 'flow', 'arch'] 

75 

76 def __init__(self, name=None, args_f=dict(), **kwargs): 

77 """ 

78 Instantiate the function superclass with the relevant parameters. 

79 

80 Parameters 

81 ---------- 

82 args_f : dict, optional 

83 arguments to pass to custom __init__ function. 

84 """ 

85 super().__init__(name=name, **kwargs) 

86 self.args_f = args_f 

87 if hasattr(self, 'behavior'): 

88 raise Exception("Invalid behavioral method: behavior(). Behavior must be" 

89 " specified as dynamic_behavior() or static_behavior()") 

90 if hasattr(self, 'condfaults'): 

91 raise Exception("Use of condfaults() is deprecated.") 

92 

93 def base_type(self): 

94 """Return fmdtools type of the model class.""" 

95 return Function 

96 

97 def return_faultmodes(self): 

98 """ 

99 Get the fault modes present in the simulation (for propagate/model). 

100 

101 Returns 

102 ------- 

103 ms : list 

104 List of faults present. 

105 modeprops : dict 

106 Dict of corresponding fault mode properties. 

107 """ 

108 if hasattr(self, 'm'): 

109 ms = [m for m in self.m.faults.copy() if m != 'nom'] 

110 modeprops = dict.fromkeys(ms) 

111 for mode in ms: 

112 modeprops[mode] = self.m.faultmodes.get(mode) 

113 if mode not in self.m.faultmodes: 

114 raise Exception("Mode " + mode + " not in m.faultmodes for fxn " + 

115 self.__class__.__name__+" and may not be tracked.") 

116 else: 

117 ms = [] 

118 modeprops = {} 

119 return ms, modeprops

121 def update_seed(self, seed=[]): 

122 """ 

123 Update seed and propogates update to contained actions/components. 

124 

125 (keeps seeds in sync) 

126 

127 Parameters 

128 ---------- 

129 seed : int, optional 

130 Random seed. The default is []. 

131 """ 

132 super().update_seed(seed) 

133 for at in self.get_roles('arch'): 

134 arch = getattr(self, at) 

135 if hasattr(arch, 'r'): 

136 arch.update_seed(self.r.seed) 

137 

138 def prop_arch_behaviors(self, proptype, faults, time, run_stochastic): 

139 """Propagate behaviors into contained architectures.""" 

140 for objname in self.get_roles('arch'): 

141 try: 

142 obj = getattr(self, objname) 

143 obj.inject_faults(faults) 

144 # TODO: this should be more general 

145 if objname == 'aa': 

146 obj(proptype, time, run_stochastic, self.t.dt) 

147 elif objname == 'fa': 

148 obj.propagate(proptype, time=time, run_stochastic=run_stochastic) 

149 except TypeError as e: 

150 raise Exception("Poorly specified Architecture: " 

151 + str(self.at.__class__)) from e 

152 

153 def prop_arch_faults_up(self): 

154 """Get faults from contained components and add to .m.""" 

155 for objname in self.get_roles('arch'): 

156 obj = getattr(self, objname) 

157 self.m.faults.difference_update(obj.faultmodes) 

158 self.m.faults.update(obj.get_faults()) 

159 

160 def __call__(self, proptype, faults=[], time=0, run_stochastic=False): 

161 """ 

162 Update the state of the function at a given time and injects faults. 

163 

164 Parameters 

165 ---------- 

166 proptype : str 

167 Type of propagation step to update 

168 ('static_behavior', or 'dynamic_behavior') 

169 faults : list, optional 

170 Faults to inject in the function. The default is []. 

171 time : float, optional 

172 Model time. The default is 0. 

173 run_stochastic : book 

174 Whether to run the simulation using stochastic or deterministic behavior 

175 """ 

176 if hasattr(self, 'r'): 

177 self.r.run_stochastic = run_stochastic 

178 # if there is a fault, it is instantiated 

179 if faults: 

180 self.m.add_fault(*faults) 

181 self.m.update_modestates() 

182 

183 if time > self.t.time: 

184 if hasattr(self, 'r'): 

185 self.r.update_stochastic_states() 

186 self.prop_arch_behaviors(proptype, faults, time, run_stochastic) 

187 

188 if proptype == 'static' and hasattr(self, 'static_behavior'): 

189 self.static_behavior(time) 

190 elif proptype == 'dynamic' and hasattr(self, 'dynamic_behavior') and time > self.t.time: 

191 if self.t.run_times >= 1: 

192 for i in range(self.t.run_times): 

193 self.dynamic_behavior(time) 

194 elif not Decimal(str(time)) % Decimal(str(self.t.dt)): 

195 self.dynamic_behavior(time) 

196 

197 self.prop_arch_faults_up() 

198 

199 self.t.time = time 

200 if run_stochastic == 'track_pdf': 

201 if hasattr(self, 'r'): 

202 self.r.probdens = self.r.return_probdens() 

203 if hasattr(self, 'm') and self.m.exclusive is True and len(self.m.faults) > 1: 

204 raise Exception("More than one fault present in " + self.name + 

205 "\n at t= " + str(time) + 

206 "\n faults: " + str(self.m.faults) + 

207 "\n Is the mode representation nonexclusive?") 

208 return 

209 

210 def return_probdens(self): 

211 """Get the probability density associated with FxnBlock and its archs.""" 

212 pd = super().return_probdens() 

213 for arch in self.archs: 

214 pd *= getattr(self, arch).return_probdens() 

215 return pd

218class ExampleFunction(Function): 

219 """Example Function block for testing.""" 

220 

221 __slots__ = ('exf',) 

222 container_p = ExampleParameter 

223 container_s = ExampleState 

224 container_m = ExampleMode 

225 flow_exf = ExampleFlow 

226 

227 def dynamic_behavior(self, time): 

228 """Increment x if nominal, else increment y.""" 

229 if not self.m.any_faults(): 

230 self.s.x += self.p.x 

231 else: 

232 self.s.y += self.p.y 

233 if time < 1.0: 

234 self.s.put(x=0.0, y=0.0) 

235 self.exf.s.inc(x=self.s.x) 

236 

237 def find_classification(self, scen, hist): 

238 """Classify via metric xy = s.x + s.y.""" 

239 return {"xy": self.s.x + self.s.y}

242class GenericFxn(Function): 

243 """ 

244 Generic function block. 

245 

246 For use when the user has not yet defined a class for the given (to be implemented) 

247 function block. Acts as a placeholder that enables simulation. 

248 """ 

249 __slots__ = ['__dict__'] 

250 check_dict_creation = False 

251 

252 def __init__(self, name='', flows={}, args_f={}, **kwargs): 

253 super().__init__(name=name, flows={}, **kwargs) 

254 self.flows = tuple([*flows.keys()]) 

255 for fl, flobj in flows.items(): 

256 setattr(self, fl, flobj)
